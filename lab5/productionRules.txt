declaration:
	functionDeclaration
	| assignment
	;

statements:
	statement semis
	| statement semis statements
	| SEMICOLON
	;

semis:
	SEMICOLON semis
	| SEMICOLON
	;

statement:
	declaration
	| loopStatement
	| expression
	;

assignment:
	IDENTIFIER ASSIGNMENT expression
	| IDENTIFIER ADD_ASSIGNMENT expression
	| IDENTIFIER SUB_ASSIGNMENT expression
	| arrayIndex ASSIGNMENT expression
	| arrayIndex ADD_ASSIGNMENT expression
	| arrayIndex SUB_ASSIGNMENT expression
	;

arrayIndex:
	IDENTIFER LSQUARE INTEGER_LITERAL RSQUARE
	;

loopStatement:
	forstatement
	| whileStatement
	| doWhileStatement
	;


expression:
	disjunction
	;

disjunction:
	disjunction DISJ conjunction
	| conjunction
	;

conjunction:
	conjunction CONJ equality
	| equality
	;

controlStructureBody:
	block
	| statement SEMICOLON
	;
